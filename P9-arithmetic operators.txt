/* arithmetic operators in c*/
#include<stdio.h>
int main()
{
    int a,b;
    float c;
    a=5;
    b=2;
    c= a+b;
    printf("sum of %d and %d is %.2f\n",a,b,c);
    c=a*b;
    printf("product of %d and %d is %.2f\n,a,b,c");
    c=a/b;
    printf("integer division of %d by %d is %.2f\n",a,b,c);
    c= (float)a/b;
    printf("real division of %d by %d is %.2f\n",a,b,c);
    return 0;
}
